package piano;

import processing.core.PApplet;
import processing.core.PImage;

public class PlayPauseButton extends Button {

    /**
     * whether the app is currently playing (true) or is paused/stopped (false)
     */
    boolean playing;

    /**
     * Constructor for PlayPauseButton
     * @param images ImageBuilder object
     */
    public PlayPauseButton(ImagesBuilder images) { // for images, sprite = play button, alternative sprite = pause button
        super(images, 5, 5);
        this.playing = false;
    }

    public void onClick(MidiHandler handler, Blocks blocks) {
        if (handler == null || blocks == null) {
            throw new NullPointerException();
        }
        this.playing = !this.playing;
        if (this.playing) {
            handler.playMidi();
        } else {
            handler.pause();
        }
    }

    public void draw(PApplet app) {
        app.image(this.images.background, this.x, this.y);
        if (this.playing) {
            app.image(this.images.alternativeSprite, this.x, this.y);
        } else {
            app.image(this.images.sprite, this.x, this.y);
        }
    }
    
    /**
     * create ImageBuilder
     * @param app current instance of app
     * @param background background image
     * @return ImagesBuilder object
     */
    public static ImagesBuilder createImageObject(PApplet app, PImage background) {
        PImage play = app.loadImage("src/main/resources/play.png");
        PImage pause = app.loadImage("src/main/resources/pause.png");
        ImagesBuilder playImages = new ImagesBuilder()
                                .setBackground(background)
                                .setSprite(play)
                                .setAlternativeSprite(pause);
        return playImages;
    }

}
