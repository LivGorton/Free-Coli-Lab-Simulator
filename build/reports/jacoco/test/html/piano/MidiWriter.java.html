package piano;

import java.io.File;

import javax.sound.midi.*;
import java.util.Random;

import processing.core.PApplet;
import processing.core.PImage;

public class MidiWriter extends Button {

    /**
     * Constructor for MidiWriter
     * @param images ImagesBuilder object
     */
    public MidiWriter(ImagesBuilder images) {
        super(images, 230, 5);
    }

    /**
     * Writes the given sequence to a .mid file
     * @param sequence the sequence to write to the .mid file
     */
    public static void writeMidi(Sequence sequence) {
        String fileName = MidiWriter.generateRandomFileName() + ".mid";
        try {
            File file = new File(fileName);
            MidiSystem.write(sequence, 0, file);
        } catch (Exception e) {
            System.err.println("file not found");
        }
    }

	public void onClick(MidiHandler handler, Blocks block) {
        MidiWriter.writeMidi(handler.getSequence());
    }

    public void draw(PApplet app) {
        app.image(this.images.background, this.x, this.y);
        app.image(this.images.sprite, this.x, this.y);
    }

    /**
     * creates ImagesBuilder object
     * @param app current instance of App
     * @param background background image
     * @return ImagesBuilder object
     */
    public static ImagesBuilder createImageObject(PApplet app, PImage background) {
        PImage sprite = app.loadImage("src/main/resources/export.png");
        ImagesBuilder images = new ImagesBuilder().setBackground(background).setSprite(sprite);
        return images;
    }

    /**
     * Generates a random 10 character string for the .mid file name.
     * Allows for a lower chance of filename collision so that the files 
     * generally persist between sessions.
     * @return the generated random string
     */
    public static String generateRandomFileName() {
        final String characters = "abcdefghijklmnopqrstuvwxyz";
        Random random = new Random();
        int min = 0;
        int max = 25;
        String randomString = "";

        for (int i = 0; i < 10; i++) {
            randomString += characters.charAt(random.nextInt(max-min+1)+min);
        }

        return randomString;
    }

}
