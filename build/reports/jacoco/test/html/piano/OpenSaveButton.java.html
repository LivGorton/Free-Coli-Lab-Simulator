package piano;

import java.io.File;
import java.util.Scanner;

import processing.core.PApplet;
import processing.core.PImage;

public class OpenSaveButton extends Button {

    /**
     * Constructor for OpenSaveButton
     * @param images ImagesBuilder object with images 
     * needed to render OpenSaveButton
     */
    public OpenSaveButton(ImagesBuilder images) {
        super(images, 185, 5);
    }

    /**
     * create ImageBuilder
     * @param app current instance of app
     * @param background background image
     * @return ImagesBuilder object
     */
    public static ImagesBuilder createImageObject(PApplet app, PImage background) {
        PImage load = app.loadImage("src/main/resources/load.png");
        ImagesBuilder images = new ImagesBuilder().setBackground(background).setSprite(load);
        return images;
    }

    /**
     * Opens the save file
     * @param handler
     * @return array the same size as Blocks array with true at indexes where 
     * notes should be active and false at indexes where notes shouldn't be active, 
     * returns null if no save is present
     */
    public boolean[][] openSave(MidiHandler handler) {

        File file = new File("save.txt");
        int lines = 0;
        int instrumentChoice = 0;
        boolean[][] notesPresent = new boolean[13][32];
        try {
            Scanner sc = new Scanner(file);
            while (sc.hasNext()) {
                lines++;
                if (lines == 1) {
                    instrumentChoice = Integer.parseInt(sc.nextLine().replace("\n", ""));
                    handler.setInstrument(instrumentChoice);
                } else {
                    int count = 0;
                    for (char c : sc.nextLine().toCharArray()) {
                        if (c == '0') {
                            notesPresent[lines - 2][count] = false;
                        } else {
                            notesPresent[lines - 2][count] = true;
                        }
                        count++;
                    }
                }
            }
            sc.close();
            return notesPresent;
        } catch (Exception e) {
            System.err.println(e);
        }
        return null;
    }

    public void onClick(MidiHandler handler, Blocks blocks) {
        boolean[][] data = this.openSave(handler);
        blocks.saveDataToBlockArray(data, handler);
        handler.stop();
    }

    public void draw(PApplet app) {
        app.image(this.images.background, this.x, this.y);
        app.image(this.images.sprite, this.x, this.y);
    }

}
