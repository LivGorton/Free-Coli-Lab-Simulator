package piano;

import processing.core.PImage;

public class ImagesBuilder {

    /**
     * Image object for the background image
     */
    PImage background;

    /**
     * Image object for the main sprite image
     */
    PImage sprite;

    /**
     * Image object for the alternative sprite image
     */
    PImage alternativeSprite;

    /**
     * Sets the background image
     * @param background the background image
     * @return ImagesBuilder object
     */
    public ImagesBuilder setBackground(PImage background) {
        this.background = background;
        return this;
    }

    /**
     * Sets the main sprite image
     * @param sprite the sprite image
     * @return ImagesBuilder object
     */
    public ImagesBuilder setSprite(PImage sprite) {
        this.sprite = sprite;
        return this;
    }

    /**
     * Sets the alternative sprite image
     * @param alternativeSprite the alternative sprite image
     * @return ImagesBuilder object
     */
    public ImagesBuilder setAlternativeSprite(PImage alternativeSprite) {
        this.alternativeSprite = alternativeSprite;
        return this;
    }

}
