package piano;

import javax.sound.midi.*;

public class MidiHandler {

    private Sequence sequence;
    private Sequencer sequencer;
    private Track track;
    private MidiEvent[][] noteOnEvents;
    private MidiEvent[][] noteOffEvents;
    private final int[] instruments = {0, 105, 12, 65};
    private int instrumentIndex;

    /**
     * Constructor for MidiHandler
     */
    public MidiHandler() {
        this.noteOnEvents = new MidiEvent[13][32];
        this.noteOffEvents = new MidiEvent[13][32];
        this.instrumentIndex = 0;
        try {
            this.sequencer = MidiSystem.getSequencer();
            this.sequence = new Sequence(Sequence.PPQ, 4);
            this.track = this.sequence.createTrack();
            this.sequencer.open();
        } catch (Exception e) {}
    }
    
    /**
     * Plays current midi track
     */
    public void playMidi() {
        try {
            this.sequencer.setTempoInBPM(120);
            this.sequencer.setSequence(this.sequence);
            this.sequencer.start();
        } catch (Exception e) {
            System.out.println(e);
        }
    }

    /**
     * Updates the position in track when it is necessary to loop around again
     * @param cursorPosition current position of the cursor
     * @param playing whether the track should be playing or not
     */
    public void updatePositionInTrack(float cursorPosition, boolean playing) {
        if (cursorPosition == 60) {
            this.sequencer.setTickPosition(0);
            if (playing) {
                this.playMidi();
            }
        }
    }

    /**
     * Stops current playback and resets position
     */
    public void stop() {
        this.sequencer.stop();
        this.sequencer.setTickPosition(0);
    }

    /**
     * Stops current playback
     */
    public void pause() {
        this.sequencer.stop();
    }

    /**
     * Add a MidiEvent to the track
     * Add that MidiEvent to noteOnEvents and noteOffEvents arrays in form array[i][j]
     * @param i vertical index in the noteOnEvents and noteOffEvents for the current event
     * @param j horizontal index in the noteOnEvents and noteOffEvents for the current event
     */
    public void addEvent(int i, int j) {
        try {
            ShortMessage noteOnMessage = new ShortMessage();
            noteOnMessage.setMessage(Notes.NOTE_ON, 0, Notes.indexToNotesMap.get(i), 100);
            MidiEvent noteOnEvent = new MidiEvent(noteOnMessage, j*2);
            this.track.add(noteOnEvent);

            ShortMessage noteOffMessage = new ShortMessage();
            noteOffMessage.setMessage(Notes.NOTE_OFF, 0, Notes.indexToNotesMap.get(i), 100);
            MidiEvent noteOffEvent = new MidiEvent(noteOffMessage, (j*2)+2);
            this.track.add(noteOffEvent);

            this.noteOnEvents[i][j] = noteOnEvent;
            this.noteOffEvents[i][j] = noteOffEvent;

        } catch (Exception e) {
            System.out.println(e);
        }
    }

    /**
     * Remove a MidiEvent to the track
     * Remove that MidiEvent to noteOnEvents and noteOffEvents arrays in form array[i][j]
     * @param i vertical index in the noteOnEvents and noteOffEvents for the current event
     * @param j horizontal index in the noteOnEvents and noteOffEvents for the current event
     */
    public void removeEvent(int i, int j) {
        if (this.noteOnEvents[i][j] != null && this.noteOffEvents[i][j] != null) {
            this.track.remove(this.noteOnEvents[i][j]);
            this.track.remove(this.noteOffEvents[i][j]);
            this.noteOnEvents[i][j] = null;
            this.noteOffEvents[i][j] = null;
        }
    }

    /**
     * Gets the sequence
     * @return the current sequence
     */
    public Sequence getSequence() {
        return this.sequence;
    }

    /**
     * Sets the current instrument to the instrument at the provided index in the instruments array
     * @param index index of the new instrument
     */
    public void setInstrument(int index) {
        if (!this.validIndex(index)) {
            throw new IndexOutOfBoundsException();
        }
        try {
            ShortMessage changeInstrument = new ShortMessage();
            changeInstrument.setMessage(ShortMessage.PROGRAM_CHANGE, 0, instruments[index], 0);
            MidiEvent instrumentChange = new MidiEvent(changeInstrument, 0);
            this.instrumentIndex = index;
            this.track.add(instrumentChange);
        } catch (Exception e) {
            System.err.println(e);
        }
    }

    /**
     * Gets the current instrument
     * @return the index of the current instrument
     */
    public int getInstrument() {
        return this.instrumentIndex;
    }

    /**
     * Gets the current array of noteOn MidiEvents
     * @return current noteOnEvents array
     */
    public MidiEvent[][] getNoteOnEvents() {
        return this.noteOnEvents;
    }

    /**
     * Gets the current array of noteOff MidiEvents
     * @return current noteOffEvents array
     */
    public MidiEvent[][] getNoteOffEvents() {
        return this.noteOffEvents;
    }

    /**
     * Checks whether a given instrument index is valid
     * @param index the instrument index
     * @return whether it is is valid
     */
    public boolean validIndex(int index) {
        if (index < 0 || index > 3) {
            return false;
        }
        return true;
    }

    // used just for testing, hence no Javadoc
    public boolean playing() {
        return this.sequencer.isRunning();
    }

}
