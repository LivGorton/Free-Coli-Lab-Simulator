package piano;

import processing.core.PApplet;
import processing.core.PImage;

public class Cursor {

    private float xPosLine;
    private float xPosCursorHead;
    private final float Y_POS;
    private final int CURSOR_LINE_WIDTH;
    private final int CURSOR_LINE_Y_POS;
    private PImage sprite;
    

    /**
     * Constructor for Cursor
     * @param app current instance of App
     */
    public Cursor(PApplet app) {
        this.sprite = app.loadImage("src/main/resources/pointer.png");
        this.xPosLine = 60;
        this.xPosCursorHead = this.xPosLine - 11;
        this.CURSOR_LINE_WIDTH = 1;
        this.CURSOR_LINE_Y_POS = 260;
        this.Y_POS = 75;
        this.setCursorColours(app);
    }

    /**
     * Sets the colours for drawing the cursor
     * @param app current instance of App
     */
    public void setCursorColours(PApplet app) {
        app.stroke(255, 0, 0);
        app.fill(255, 0, 0);
    }

    /**
     * Moves the cursor along when App is in the playing state
     * @param playing whether the App is currently in 
     * the active, playing state
     */
    public void move(boolean playing) {
        if (playing) {
            this.xPosLine++;
            this.xPosCursorHead++;
            if (this.xPosLine > 540) {
                this.resetCursorPosition();
            }
        }
    }
    /**
     * resets the cursor position when it needs to 
     * loop back to the start of the display
     */
    public void resetCursorPosition() {
        this.xPosLine = 60;
        this.xPosCursorHead = this.xPosLine - 11;
    }

    /**
     * Gets the current xPosition
     * @return the current x position of the cursor
     */
    public float getXPosition() {
        return this.xPosLine;
    }

    /**
     * Gets the YPosition
     * @return the unchanging y position of cursor
     */
    public float getYPositon() {
        return this.Y_POS;
    }

    /**
     * Draw the cursor 
     * @param app current instance of App
     */
    public void draw(PApplet app) {     
        app.rect(this.xPosLine, this.Y_POS, this.CURSOR_LINE_WIDTH, this.CURSOR_LINE_Y_POS);
        app.image(this.sprite, this.xPosCursorHead, this.Y_POS);
    }
}
