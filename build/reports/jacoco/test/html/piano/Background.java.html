package piano;

import processing.core.PApplet;
import processing.core.PImage;

public class Background {

    private static PImage keyboard;
    private static PImage grid;

    /**
     * Constructor for Background. Loads the images that will
     * need to be re-rendered so that they are only loaded
     * one. Requires an instance of App so it can load and display images. 
     * @param app the current instance of App
     */
    public Background(PApplet app) {
        keyboard = app.loadImage("src/main/resources/keyboard.png");
        grid = app.loadImage("src/main/resources/grid.png");
    }

    /**
     * Sets up the background images that won't need to be reloaded
     * each frame. Requires an instance of App so it can load and display images
     * @param app the current instance of App
     */
    public static void setupBackground(PApplet app) {
        PImage middleBanner = app.loadImage("src/main/resources/middleBanner.png");
        PImage banner = app.loadImage("src/main/resources/banner.png");
        app.image(middleBanner, 0, 0);
        app.image(banner, 0, 0);
    }
    
    /**
     * Sets up the background images that will need to be reloaded
     * each frame. Requires an instance of App so it can load and display images
     * @param app the current instance of App
     */
    public static void drawDynamicBackground(PApplet app) {
        app.image(keyboard, 0, 75);
        app.image(grid, 60, 75);
    }
    
}
