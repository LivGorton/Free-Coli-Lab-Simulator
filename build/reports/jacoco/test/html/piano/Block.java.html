package piano;

import processing.core.PApplet;
import processing.core.PImage;

public class Block {

    private boolean active;
    private PImage blockSprite;
    private int x;
    private int y;
    private int indexX;
    private int indexY;

    /**
     * The constructor for the Block object
     * @param app the current instance of App
     * @param x the x coordinate of the Block image display
     * @param y the y coordinate of the Block image display
     * @param indexX the x index for the Block's position 
     * in the Blocks notesPresent 2D array
     * @param indexY the y index for the Block's position 
     * in the Blocks notesPresent 2D array
     */
    public Block(PApplet app, int x, int y, int indexX, int indexY) {
        this.active = false;
        this.blockSprite = app.loadImage("src/main/resources/block.png");;
        this.x = x;
        this.y = y;
        this.indexX = indexX;
        this.indexY = indexY;
    }

    /**
     * Activates and deactivates the block 
     * when the block has been pressed
     * @param handler an instance of MidiHandler
     */
    public void click(MidiHandler handler) {
        this.active = !this.active;
        if (this.active) {
            handler.addEvent(this.indexY, this.indexX);
        } else {
            handler.removeEvent(this.indexY, this.indexX);
        }
    }

    /**
     * Gets the current state of the block
     * @return if the block is currently active or not
     */
    public boolean isActive() {
        return this.active;
    }

    /**
     * Draws a single block
     * @param app the current instance of App
     */
    public void drawBlock(PApplet app) {
        if (this.active) {
            app.image(this.blockSprite, this.x, this.y);
        }
    }

    /**
     * Deactivates a block regardless of its previous state
     * @param handler the current instance of MidiHandler
     */
    public void deactivate(MidiHandler handler) {
        this.active = false;
        handler.removeEvent(this.indexY, this.indexX);
    }

}
