package piano;

import processing.core.PApplet;
import processing.core.PImage;
import java.io.*;

public class SaveButton extends Button {

    /**
     * Constructor for SaveButton
     * @param images ImagesBuilder object
     */
    public SaveButton(ImagesBuilder images) {
        super(images, 140, 5);
    }

    /**
     * Saves the current set of Block states and the current instrument to a txt file
     * @param handler the current instance of MidiHandler
     * @param blocks the current instance of Blocks
     */
    public void save(MidiHandler handler, Blocks blocks) {
        try {
            File file = new File("save.txt");
            FileWriter fw = new FileWriter(file, false);
            fw.write(Integer.toString(handler.getInstrument()) + '\n');
            for (int i = 0; i < blocks.getBlocks().length; i++) {
                String currentValues = "";
                for (int j = 0; j < blocks.getBlocks()[0].length; j++) {
                    if (blocks.getBlocks()[i][j].isActive()) {
                        currentValues += '1';
                    } else {
                        currentValues += '0';
                    }
                }
                fw.write(currentValues + '\n');
            }
            fw.close();
        } catch(Exception e) {
            System.err.println(e);
        }
    }

    public void onClick(MidiHandler handler, Blocks blocks) {
        this.save(handler, blocks);
    }

    public void draw(PApplet app) {
        app.image(this.images.background, this.x, this.y);
        app.image(this.images.sprite, this.x, this.y);
    }

    /**
     * create ImageBuilder
     * @param app current instance of app
     * @param background background image
     * @return ImagesBuilder object
     */
    public static ImagesBuilder createImageObject(PApplet app, PImage background) {
        PImage load = app.loadImage("src/main/resources/save.png");
        ImagesBuilder images = new ImagesBuilder()
                                .setBackground(background)
                                .setSprite(load);
        return images;
    }
}
