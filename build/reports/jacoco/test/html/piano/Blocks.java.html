package piano;

import processing.core.PApplet;

public class Blocks {

    private Block[][] blocks;

    /**
     * Constructor for Blocks
     */
    public Blocks() {
        this.blocks = new Block[13][32];
    }

    /**
     * Gets the current array of Block objects
     * @return the Block[][] array
     */
    public Block[][] getBlocks() {
        return this.blocks;
    }

    /**
     * Clicks on a specific block
     * @param x x coordinate in Block[][] array
     * @param y y coordinate in Block[][] array
     * @param handler current instance of MidiHandler
     */
    public void setBlock(int x, int y, MidiHandler handler) {
        this.blocks[y][x].click(handler);
    }

    /**
     * Populates the blocks array with deactivated Block objects
     * @param app current instance of App
     */
    public void populateBlocks(PApplet app) {
        for (int i = 0; i < this.blocks.length; i++) {
            for (int j = 0; j < this.blocks[0].length; j++) {
                int y =  20*i+75;
                int x = 15*j+60;
                this.blocks[i][j] = new Block(app, x, y, j, i);
            }
        }
    }

    /**
     * Draws all blocks currently held by Blocks object
     * @param app current instance of App
     */
    public void drawBlocks(PApplet app) {
        for (int i = 0; i < this.blocks.length; i++ ){
            for (int j = 0; j < this.blocks[0].length; j++) {
                this.blocks[i][j].drawBlock(app);
            }
        }
    }

    /**
     * Deactivates all Block objects
     * @param handler current instance of MidiHandler
     */
    public void removeAll(MidiHandler handler) {
        for (int i = 0; i < this.blocks.length; i++) {
            for (int j = 0; j < this.blocks[0].length; j++) {
                this.blocks[i][j].deactivate(handler);
            }
        }
    }

    /**
     * Reads in an array of current state of every block
     * false for inactive, true for active
     * Resets the Blocks state to the one dictated by 
     * the notes parameter
     * @param notes an array of boolean objects
     * @param handler current instance of MidiHandler
     */
    public void saveDataToBlockArray(boolean[][] notes, MidiHandler handler) {
        if (notes != null) {
            for (int i = 0; i < this.blocks.length; i++ ){
                for (int j = 0; j < this.blocks[0].length; j++) {
                    if (notes[i][j] == true && this.blocks[i][j].isActive() == false) {
                        this.blocks[i][j].click(handler);
                    } else if (notes[i][j] == false && this.blocks[i][j].isActive() == true) {
                        this.blocks[i][j].deactivate(handler);
                    }
                }
            }
        }
    }

}
