package piano;

import processing.core.PApplet;
import processing.core.PImage;

public class App extends PApplet {
    
    /** Create MidiHandler object */ public MidiHandler handler;
    /** Create Cursor object */ public Cursor cursor;
    /** Create image for shared button background */ public PImage buttonBacking;
    /** Create PlayPauseButton object */ public PlayPauseButton playPauseButton;
    /** Create StopButton object */ public StopButton stopButton;
    /** Create ClearButton object */ public ClearButton clearButton;
    /** Create InstrumentChangeButtons object */ public InstrumentChangeButtons instrumentChangeButtons;
    /** Create Blocks object */ public Blocks blocks;
    /** Create OpenSaveButton object */ public OpenSaveButton openSaveButton;
    /** Create SaveButton object */ public SaveButton saveButton;
    /** Create MidiWriter object */ public MidiWriter midiWriter;
    /** Create Background object */ public Background background;

    /**
     * Setup non-image based objects
     */
    public App() {
        this.handler = new MidiHandler();   
    }

    /**
     * Change the window settings
     */
    public void settings() {
        // Don't touch
        size(540, 335);
    }

    /**
     * Setup the app frame rate and create objects 
     * that need images associated with them
     * The objects created are all buttons, the cursor, 
     * the blocks on the grid, and the background
     */
    public void setup() {
        // set frame rate
        frameRate(60);

        buttonBacking = loadImage("src/main/resources/buttonBack.png");
        this.cursor = new Cursor(this);
        this.blocks = new Blocks();
        this.blocks.populateBlocks(this);
        this.background = new Background(this);
        this.stopButton = new StopButton(StopButton.createImageObject(this, buttonBacking));
        this.clearButton = new ClearButton(ClearButton.createImageObject(this, buttonBacking));
        this.openSaveButton = new OpenSaveButton(OpenSaveButton.createImageObject(this, buttonBacking));
        this.saveButton = new SaveButton(SaveButton.createImageObject(this, buttonBacking));
        this.instrumentChangeButtons = new InstrumentChangeButtons(InstrumentChangeButtons.createImageObject(this, buttonBacking), 
                                                                    InstrumentChangeButtons.getImageChoices(this));
        this.midiWriter = new MidiWriter(MidiWriter.createImageObject(this, buttonBacking));
        this.playPauseButton = new PlayPauseButton(PlayPauseButton.createImageObject(this, buttonBacking));
        // display background that don't need to be re-rendered every frame, load the other background images, and draw buttons
        Background.setupBackground(this);
        this.stopButton.draw(this);
        this.clearButton.draw(this);
        this.openSaveButton.draw(this);
        this.saveButton.draw(this);
        this.midiWriter.draw(this);
        this.instrumentChangeButtons.draw(this);
    }

    /**
     * Draw the current state of the game objects
     */
    public void draw() {    
        Background.drawDynamicBackground(this); // re-draw content that might change between frames
        this.playPauseButton.draw(this);
        instrumentChangeButtons.draw(this, this.handler);
        this.blocks.drawBlocks(this);
        this.cursor.draw(this);
        this.cursor.move(playPauseButton.playing);
        // update position in midi track when it needs to loop
        this.handler.updatePositionInTrack(this.cursor.getXPosition(), playPauseButton.playing);
    }
    
    /**
     * Method is called when the mouse is pressed by the user
     * Passes current App instance to ClickHandler object which checks
     * which buttons have been pressed
     */
    public void mousePressed() { 
        ClickHandler.handleClick(mouseX, mouseY, this);
    }

    public static void main(String[] args) {
        // Don't touch this
        PApplet.main("piano.App");
    }
}
