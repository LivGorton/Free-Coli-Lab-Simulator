package piano;

import processing.core.PApplet;
import processing.core.PImage;

public class InstrumentChangeButtons extends Button {

    private int buttonMinusX;
    private int buttonMinusY;
    private int buttonPlusX;
    private int buttonPlusY;
    private PImage[] instrumentImages;

    /**
     * Constructor for InstrumentChangeButtons 
     * @param images ImagesBuilder object with images for this Button
     * @param instrumentImages array of possible images for one of the Buttons
     */
    public InstrumentChangeButtons(ImagesBuilder images, PImage[] instrumentImages) {
        // sprite = plus image, alternativeSprite = minus image
        super(images, 320, 5); //picked to use for the actual display of the instrument
        // setup up all positions of buttons/images
        this.buttonMinusY = 5;
        this.buttonPlusY = 5;
        this.buttonMinusX = 275;
        this.buttonPlusX = 365;
        this.instrumentImages = instrumentImages;
    }

    /**
     * Handles the click of all three buttons in collection
     * @param mouseX mouse X position, dictates which of the three buttons was pressed
     * @param handler current instance of MidiHandler
     */
    public void onClick(int mouseX, MidiHandler handler) {

        if (mouseX <= 315) {
            if (handler.getInstrument() > 0) {
                handler.setInstrument(handler.getInstrument() - 1);
            }
        } else if (mouseX >= 365) {
            if (handler.getInstrument() < 3) {
                handler.setInstrument(handler.getInstrument() + 1);
            }
        }

    }

    public void draw(PApplet app) { // needed to have due to extending Button but method overload draw for this class
    
    }

    /**
     * A unique draw method for this Button collection that draws all buttons
     * @param app current instance of App
     * @param handler current instance of MidiHandler
     */
    public void draw(PApplet app, MidiHandler handler) {
        app.image(this.images.background, this.buttonMinusX, this.buttonMinusY);
        app.image(this.images.background, this.x, this.y);
        app.image(this.images.background, this.buttonPlusX, this.buttonPlusY);

        app.image(this.images.alternativeSprite, this.buttonMinusX, this.buttonMinusY);
        app.image(this.instrumentImages[handler.getInstrument()], this.x, this.y);
        app.image(this.images.sprite, this.buttonPlusX, this.buttonPlusY);
    }

    /**
     * Gets all possible instrument image choices
     * @param app current instance of App
     * @return array of possible image choices
     */
    public static PImage[] getImageChoices(PApplet app) {
        PImage p = app.loadImage("src/main/resources/P.png");
        PImage b = app.loadImage("src/main/resources/B.png");
        PImage m = app.loadImage("src/main/resources/M.png");
        PImage s = app.loadImage("src/main/resources/S.png");
        PImage[] imageChoices = {p, b, m, s};
        return imageChoices;
    }

    /**
     * create ImageBuilder
     * @param app current instance of app
     * @param background background image
     * @return ImagesBuilder object
     */
    public static ImagesBuilder createImageObject(PApplet app, PImage background) {
        PImage prev = app.loadImage("src/main/resources/prev.png");
        PImage next = app.loadImage("src/main/resources/next.png");
        ImagesBuilder changeInstrumentImages = new ImagesBuilder()
                                            .setBackground(background)
                                            .setSprite(next)
                                            .setAlternativeSprite(prev);

        return changeInstrumentImages;
    }

    public void onClick(MidiHandler handler, Blocks block) {
        // 
    }

}
