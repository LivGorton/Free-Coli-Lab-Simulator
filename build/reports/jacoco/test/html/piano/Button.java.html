package piano;

import processing.core.PApplet;

public abstract class Button {

    /**
     * ImagesBuilder object contained the images for a specific button
     */
    ImagesBuilder images;
    
    /**
     * x coordinate for button display
     */
    int x;
    
    /**
     * y coordinate for button display
     */
    int y;

    /**
     * Constructor for Button
     * @param images an ImagesBuilder object
     * @param x x coordinate for set of images to be displayed
     * @param y y coordinate for set of images to be displayed
     */
    public Button(ImagesBuilder images, int x, int y) {
        this.images = images;
        this.x = x; 
        this.y = y;
    }

    /**
     * Handles the click of the Button object
     * @param handler current instance of MidiHandler
     * @param blocks Blocks object
     */
    public abstract void onClick(MidiHandler handler, Blocks blocks);

    /**
     * Draws the current Button state
     * @param app current instance of App
     */
    public abstract void draw(PApplet app);

}
