package piano;

public class ClickHandler {

    /**
     * Handles the position of all mouse presses in the App
     * @param mouseX x coordinate of mouse click
     * @param mouseY y coordinate of mouse click
     * @param app instance of App
     */
    public static void handleClick(int mouseX, int mouseY, App app) {

        if (mouseY <= 45 && mouseY >= 5) { // handle button clicks
            if (mouseX <= 45 && mouseX >= 5) {
                app.playPauseButton.onClick(app.handler, app.blocks);
            } else if (mouseX >= 50 && mouseX <= 90) {
                app.cursor.resetCursorPosition();
                app.playPauseButton.playing = false;
                app.stopButton.onClick(app.handler, app.blocks);
            } else if (mouseX >= 95 && mouseX <= 135) {
                app.clearButton.onClick(app.handler, app.blocks);
            } else if (mouseX >= 275 && mouseX <= 405) {
                app.instrumentChangeButtons.onClick(mouseX, app.handler);
            } else if (mouseX >= 140 && mouseX <= 180) {
                app.saveButton.onClick(app.handler, app.blocks);
            } else if (mouseX >= 185 && mouseX <= 225) {
                app.openSaveButton.onClick(app.handler, app.blocks);
                app.cursor.resetCursorPosition();
                app.playPauseButton.playing = false;
            } else if (mouseX >= 230 && mouseX <= 270) {
                app.midiWriter.onClick(app.handler, app.blocks);
            }
        } else { // handle clicking of blocks
            if (mouseX <= 540 && mouseX >= 60 && mouseY >= 75) { 
                app.blocks.setBlock((int) (mouseX - 60)/15, (int) (mouseY - 75)/20, app.handler);
            }
        }
    }

}
